```go
package main

import (  
    "fmt"
)

func firstWorker() {  
    fmt.Println("run your first worker")
}

func secondWorker() {  
    fmt.Println("run your second worker")
}

func main() {  
    go firstWorker()
    go secondWorker()
    fmt.Println("main running")
}
```
